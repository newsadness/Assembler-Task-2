Задача 2

Код программы:
section .data
fmt db "Result = %d", 10, 0
x dd 20 ; Значение x
a dd 5 ; Значение a
result dd 0 ; Переменная
section .text
extern printf
global main
; Psub(x, a)(Задание)
Psub:
; Параметры передаются через стек
push ebp
mov ebp, esp
mov eax, [ebp + 8]
sub eax, [ebp + 12]
mov [result], eax
mov esp, ebp
pop ebp
ret
main:
; Psub
push dword [a]
push dword [x]
call Psub
add esp, 8
; Вывод результата через переменную
push dword [result]
push fmt
call printf
add esp, 8
; Завершение программы
mov eax, 0
ret
